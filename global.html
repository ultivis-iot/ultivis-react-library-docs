  const isDevice = useUserAgentStore((state) => state.isDevice);

  if (isDevice === null) {
    return <div>Loading...</div>;
  }

  return (
    <div>
      {isDevice ? <MobileView /> : <DesktopView />}
    </div>
  );
}
// Set user agent type during initialization
import { useUserAgentStore } from '@ultivis/react-library/stores';

function App() {
  useEffect(() => {
    const userAgent = navigator.userAgent.toLowerCase();
    const type = userAgent.includes('mobile') ? 'mobile' : 'desktop';
    useUserAgentStore.getState().setIsDevice(type);
  }, []);

  return <AppContent />;
}
