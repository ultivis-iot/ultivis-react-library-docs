üöÄ Quick Start

1. Setup AppProviders

import { AppProviders } from '@ultivis-iot/react';
import { createBrowserRouter } from 'react-router-dom';
import '@ultivis-iot/react/styles';

const router = createBrowserRouter([
  // your routes
]);

function App() {
  return (
    <AppProviders
      router={router}
      favicon="/favicon.ico"
      brandLogo="/logo.png"
      navLogo="/nav-logo.png"
      login={true}
    />
  );
}

export default App;

2. Using Layout

import { Layout } from '@ultivis-iot/react';

function MyPage() {
  return (
    <Layout
      headerItems={<></>}
      sidebarItems={<></>}
      pageTitle={() => 'My Page'}
    >
      {/* Page content */}
    </Layout>
  );
}
