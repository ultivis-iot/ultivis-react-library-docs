import { forwardRef } from 'react';
import * as NavigationMenuPrimitive from '@radix-ui/react-navigation-menu';
import { cva } from 'class-variance-authority';
import { cn } from '@utils';

/**
 * Navigation Menu System
 *
 * A comprehensive navigation menu system built on Radix UI primitives.
 * Supports hierarchical navigation with dropdown content, keyboard navigation,
 * and smooth animations.
 *
 * @see {@link https://www.radix-ui.com/docs/primitives/components/navigation-menu} - Radix UI Navigation Menu documentation
 */

/**
 * Navigation Menu Root Component
 *
 * The root container for the navigation menu system.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Menu content (NavigationMenuList with NavigationMenuItem elements).
 * @param {string} [props.className] - Additional CSS classes for the root container.
 * @param {string} [props.viewportClassNames] - Additional CSS classes for the viewport container.
 * @param {string} [props.value] - Controlled active menu item value.
 * @param {string} [props.defaultValue] - Default active menu item value (uncontrolled).
 * @param {Function} [props.onValueChange] - Callback when active item changes.
 * @param {React.Ref} ref - Forwarded ref to the root element.
 *
 * @returns {JSX.Element} Navigation menu root with integrated viewport.
 *
 * @example
 * <NavigationMenu>
 *   <NavigationMenuList>
 *     <NavigationMenuItem>
 *       <NavigationMenuTrigger>Products</NavigationMenuTrigger>
 *       <NavigationMenuContent>
 *         <div>Product links...</div>
 *       </NavigationMenuContent>
 *     </NavigationMenuItem>
 *   </NavigationMenuList>
 * </NavigationMenu>
 */
const NavigationMenu = forwardRef(
  ({ className, viewportClassNames, children, ...props }, ref) => (
    <NavigationMenuPrimitive.Root
      ref={ref}
      className={cn(
        'relative z-10 flex max-w-max flex-1 items-center justify-center',
        className
      )}
      {...props}
    >
      {children}
      <NavigationMenuViewport viewportClassNames={viewportClassNames} />
    </NavigationMenuPrimitive.Root>
  )
);
NavigationMenu.displayName = NavigationMenuPrimitive.Root.displayName;

/**
 * Navigation Menu List Component
 *
 * Container for navigation menu items with flexbox layout.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Menu items (NavigationMenuItem elements).
 * @param {string} [props.className] - Additional CSS classes.
 * @param {React.Ref} ref - Forwarded ref to the list element.
 *
 * @returns {JSX.Element} Styled navigation menu list.
 *
 * @example
 * <NavigationMenuList>
 *   <NavigationMenuItem>...</NavigationMenuItem>
 *   <NavigationMenuItem>...</NavigationMenuItem>
 * </NavigationMenuList>
 */
const NavigationMenuList = forwardRef(({ className, ...props }, ref) => (
  <NavigationMenuPrimitive.List
    ref={ref}
    className={cn(
      'group flex flex-1 list-none items-center justify-center space-x-1',
      className
    )}
    {...props}
  />
));
NavigationMenuList.displayName = NavigationMenuPrimitive.List.displayName;

/**
 * Navigation Menu Item Component
 *
 * Individual menu item that can contain a trigger and content.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Item content (NavigationMenuTrigger and/or NavigationMenuContent).
 * @param {string} [props.value] - Unique value for this menu item.
 *
 * @example
 * <NavigationMenuItem value="products">
 *   <NavigationMenuTrigger>Products</NavigationMenuTrigger>
 *   <NavigationMenuContent>...</NavigationMenuContent>
 * </NavigationMenuItem>
 */
const NavigationMenuItem = NavigationMenuPrimitive.Item;

/**
 * Navigation Menu Trigger Style Variant
 *
 * Class variance authority configuration for navigation menu trigger styling.
 * Can be used to style custom trigger elements or links.
 *
 * @constant
 *
 * @example
 * <NavigationMenuLink className={navigationMenuTriggerStyle()}>
 *   Home
 * </NavigationMenuLink>
 */
const navigationMenuTriggerStyle = cva(
  'group inline-flex h-10 w-max items-center justify-center rounded-md px-4 py-2 text-sm font-medium transition-colors hover:bg-transparent hover:text-accent-foreground focus:bg-transparent focus:text-accent-foreground focus:outline-none disabled:pointer-events-none disabled:opacity-50 data-[active]:bg-transparent date-[active]:text-primary data-[state=open]:bg-transparent'
);

/**
 * Navigation Menu Trigger Component
 *
 * Button that opens the dropdown content for a menu item.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Trigger button content (text and optional icon).
 * @param {string} [props.className] - Additional CSS classes.
 * @param {React.Ref} ref - Forwarded ref to the trigger element.
 *
 * @returns {JSX.Element} Styled trigger button.
 *
 * @example
 * <NavigationMenuTrigger>Products</NavigationMenuTrigger>
 *
 * @example
 * // With custom styling
 * <NavigationMenuTrigger className="text-primary">
 *   Features
 * </NavigationMenuTrigger>
 */
const NavigationMenuTrigger = forwardRef(
  ({ className, children, ...props }, ref) => (
    <NavigationMenuPrimitive.Trigger
      ref={ref}
      className={cn(navigationMenuTriggerStyle(), 'group', className)}
      {...props}
    >
      {children}
      {/* <ChevronDown
        className="relative top-[1px] ml-1 h-3 w-3 transition duration-200 group-data-[state=open]:rotate-180"
        aria-hidden="true"
      /> */}
    </NavigationMenuPrimitive.Trigger>
  )
);
NavigationMenuTrigger.displayName = NavigationMenuPrimitive.Trigger.displayName;

/**
 * Navigation Menu Content Component
 *
 * Dropdown content displayed when a menu trigger is activated.
 * Supports smooth animations and directional slide effects.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Content to display in the dropdown.
 * @param {string} [props.className] - Additional CSS classes.
 * @param {React.Ref} ref - Forwarded ref to the content element.
 *
 * @returns {JSX.Element} Animated dropdown content container.
 *
 * @example
 * <NavigationMenuContent>
 *   <div className="grid gap-3 p-6">
 *     <a href="/product-1">Product 1</a>
 *     <a href="/product-2">Product 2</a>
 *   </div>
 * </NavigationMenuContent>
 */
const NavigationMenuContent = forwardRef(({ className, ...props }, ref) => (
  <NavigationMenuPrimitive.Content
    ref={ref}
    className={cn(
      'data-[motion^=from-]:animate-in data-[motion^=to-]:animate-out data-[motion^=from-]:fade-in data-[motion^=to-]:fade-out data-[motion=from-end]:slide-in-from-right-52 data-[motion=from-start]:slide-in-from-left-52 data-[motion=to-end]:slide-out-to-right-52 data-[motion=to-start]:slide-out-to-left-52 left-0 top-0 w-full sm:absolute sm:w-auto ',
      className
    )}
    {...props}
  />
));
NavigationMenuContent.displayName = NavigationMenuPrimitive.Content.displayName;

/**
 * Navigation Menu Link Component
 *
 * Link element for navigation menu items with proper accessibility.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Link content.
 * @param {string} [props.href] - Link destination URL.
 * @param {boolean} [props.active] - Whether this link is currently active.
 * @param {Function} [props.onSelect] - Callback when link is selected.
 *
 * @example
 * <NavigationMenuLink href="/" active>
 *   Home
 * </NavigationMenuLink>
 *
 * @example
 * // With custom trigger styling
 * <NavigationMenuLink
 *   href="/about"
 *   className={navigationMenuTriggerStyle()}
 * >
 *   About Us
 * </NavigationMenuLink>
 */
const NavigationMenuLink = NavigationMenuPrimitive.Link;

/**
 * Navigation Menu Viewport Component
 *
 * Container that displays the active menu content with smooth transitions.
 * Automatically positioned and sized based on the content.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {string} [props.className] - Additional CSS classes for the viewport.
 * @param {string} [props.viewportClassNames] - Additional CSS classes for the viewport wrapper.
 * @param {React.Ref} ref - Forwarded ref to the viewport element.
 *
 * @returns {JSX.Element} Positioned viewport container with animations.
 *
 * @example
 * // Usually used automatically within NavigationMenu
 * // Can be customized with viewportClassNames prop on NavigationMenu
 * <NavigationMenu viewportClassNames="top-16">
 *   ...
 * </NavigationMenu>
 */
const NavigationMenuViewport = forwardRef(
  ({ className, viewportClassNames, ...props }, ref) => (
    <div
      className={cn(
        `absolute left-auto right-0 top-full flex justify-center ${viewportClassNames}`
      )}
    >
      <NavigationMenuPrimitive.Viewport
        className={cn(
          'origin-top-center grayscale-400 data-[state=open]:animate-in data-[state=closed]:animate-out data-[state=closed]:zoom-out-95 data-[state=open]:zoom-in-90 relative mt-1.5 h-[var(--radix-navigation-menu-viewport-height)] w-full overflow-hidden rounded-none border bg-popover text-popover-foreground shadow-lg dark:bg-dark-bg-444 sm:w-[var(--radix-navigation-menu-viewport-width)]',
          className
        )}
        ref={ref}
        {...props}
      />
    </div>
  )
);
NavigationMenuViewport.displayName =
  NavigationMenuPrimitive.Viewport.displayName;

/**
 * Navigation Menu Indicator Component
 *
 * Visual indicator that shows which menu item is currently active.
 * Displays an arrow or marker below the active trigger.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {string} [props.className] - Additional CSS classes.
 * @param {React.Ref} ref - Forwarded ref to the indicator element.
 *
 * @returns {JSX.Element} Animated indicator with arrow shape.
 *
 * @example
 * <NavigationMenuItem>
 *   <NavigationMenuTrigger>Products</NavigationMenuTrigger>
 *   <NavigationMenuContent>...</NavigationMenuContent>
 *   <NavigationMenuIndicator />
 * </NavigationMenuItem>
 */
const NavigationMenuIndicator = forwardRef(({ className, ...props }, ref) => (
  <NavigationMenuPrimitive.Indicator
    ref={ref}
    className={cn(
      'data-[state=visible]:animate-in data-[state=hidden]:animate-out data-[state=hidden]:fade-out data-[state=visible]:fade-in top-full z-[49] flex h-1.5 items-end justify-center overflow-hidden',
      className
    )}
    {...props}
  >
    <div className="relative top-[60%] h-2 w-2 rotate-45 rounded-tl-sm bg-border shadow-md" />
  </NavigationMenuPrimitive.Indicator>
));
NavigationMenuIndicator.displayName =
  NavigationMenuPrimitive.Indicator.displayName;

export {
  navigationMenuTriggerStyle,
  NavigationMenu,
  NavigationMenuList,
  NavigationMenuItem,
  NavigationMenuContent,
  NavigationMenuTrigger,
  NavigationMenuLink,
  NavigationMenuIndicator,
  NavigationMenuViewport,
};
