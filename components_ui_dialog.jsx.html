import { forwardRef } from 'react';
import * as DialogPrimitive from '@radix-ui/react-dialog';
import { X } from 'lucide-react';
import { cn } from '@utils';
import { useTranslation } from '@hooks';

/**
 * Dialog System
 *
 * A modal dialog system built on Radix UI primitives with customizable overlay,
 * close button, and outside click behavior. Provides accessible modal dialogs
 * with proper focus management and keyboard navigation.
 *
 * @see {@link https://www.radix-ui.com/docs/primitives/components/dialog} - Radix UI Dialog documentation
 */

/**
 * Dialog Root Component
 *
 * The root container for the dialog system. Controls the open/closed state.
 *
 * @component
 *
 * @param {boolean} [open] - Controlled open state of the dialog.
 * @param {boolean} [defaultOpen] - Default open state (uncontrolled).
 * @param {Function} [onOpenChange] - Callback fired when open state changes.
 * @param {boolean} [modal=true] - Whether the dialog is modal (traps focus).
 *
 * @example
 * <Dialog>
 *   <DialogTrigger>Open Dialog</DialogTrigger>
 *   <DialogContent>
 *     <DialogHeader>
 *       <DialogTitle>Dialog Title</DialogTitle>
 *       <DialogDescription>Dialog description</DialogDescription>
 *     </DialogHeader>
 *   </DialogContent>
 * </Dialog>
 */
const Dialog = DialogPrimitive.Root;

/**
 * Dialog Trigger Component
 *
 * Button that triggers opening the dialog.
 *
 * @component
 *
 * @param {React.ReactNode} children - Trigger button content.
 * @param {string} [className] - Additional CSS classes.
 * @param {boolean} [asChild=false] - Merge props with child element.
 *
 * @example
 * <DialogTrigger asChild>
 *   <Button>Open</Button>
 * </DialogTrigger>
 */
const DialogTrigger = DialogPrimitive.Trigger;

/**
 * Dialog Portal Component
 *
 * Renders dialog content in a portal at the end of the document body.
 *
 * @component
 *
 * @param {React.ReactNode} children - Dialog content to render in portal.
 * @param {HTMLElement} [container] - Custom container element for the portal.
 *
 * @example
 * <DialogPortal>
 *   <DialogOverlay />
 *   <DialogContent>...</DialogContent>
 * </DialogPortal>
 */
const DialogPortal = DialogPrimitive.Portal;

/**
 * Dialog Close Component
 *
 * Button that closes the dialog when clicked.
 *
 * @component
 *
 * @param {React.ReactNode} children - Close button content.
 * @param {string} [className] - Additional CSS classes.
 * @param {boolean} [asChild=false] - Merge props with child element.
 *
 * @example
 * <DialogClose asChild>
 *   <Button variant="outline">Cancel</Button>
 * </DialogClose>
 */
const DialogClose = DialogPrimitive.Close;

/**
 * Dialog Overlay Component
 *
 * Semi-transparent backdrop overlay behind the dialog.
 *
 * @component
 *
 * @param {string} [className] - Additional CSS classes for the overlay.
 * @param {React.Ref} ref - Forwarded ref to the overlay element.
 * @param {Object} props - Additional props passed to the overlay.
 *
 * @returns {JSX.Element} Animated backdrop overlay element.
 *
 * @example
 * <DialogOverlay className="bg-black/80" />
 */
const DialogOverlay = forwardRef(({ className, ...props }, ref) => (
  <DialogPrimitive.Overlay
    ref={ref}
    className={cn(
      'data-[state=open]:animate-in data-[state=closed]:animate-out data-[state=closed]:fade-out-0 data-[state=open]:fade-in-0 fixed inset-0 z-50 bg-grayscale-100/70 backdrop-blur-sm',
      className
    )}
    {...props}
  />
));
DialogOverlay.displayName = DialogPrimitive.Overlay.displayName;

/**
 * Dialog Content Component
 *
 * The main content container for the dialog with optional overlay and close button.
 * Supports customizable outside click behavior and animations.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Dialog content.
 * @param {string} [props.className] - Additional CSS classes for the content.
 * @param {string} [props.overlayClassName] - Additional CSS classes for the overlay.
 * @param {boolean} [props.overlay=true] - Whether to show the backdrop overlay.
 * @param {boolean} [props.closeButton=true] - Whether to show the close button in the header.
 * @param {boolean} [props.disableOutsideClick=true] - When true, clicking outside won't close the dialog.
 * @param {React.Ref} ref - Forwarded ref to the content element.
 *
 * @returns {JSX.Element} The dialog content with portal, overlay, and close button.
 *
 * @example
 * // Basic dialog
 * <DialogContent>
 *   <p>Dialog content here</p>
 * </DialogContent>
 *
 * @example
 * // Without overlay or close button
 * <DialogContent overlay={false} closeButton={false}>
 *   <p>Minimal dialog</p>
 * </DialogContent>
 *
 * @example
 * // Allow outside click to close
 * <DialogContent disableOutsideClick={false}>
 *   <p>Click outside to close</p>
 * </DialogContent>
 */
const DialogContent = forwardRef(
  (
    {
      className,
      overlayClassName = '',
      children,
      overlay = true,
      closeButton = true,
      disableOutsideClick = true,
      ...props
    },
    ref
  ) => {
    const { t } = useTranslation();
    return (
      <DialogPortal>
        {overlay && <DialogOverlay className={overlayClassName} />}
        <DialogPrimitive.Content
          ref={ref}
          // 외부 클릭 시 Dialog 닫히지않게 수정
          onInteractOutside={(e) => {
            if (disableOutsideClick) {
              e.preventDefault();
            }
          }}
          className={cn(
            'data-[state=open]:animate-in data-[state=closed]:animate-out data-[state=closed]:fade-out-0 data-[state=open]:fade-in-0 data-[state=closed]:zoom-out-95 data-[state=open]:zoom-in-95 data-[state=closed]:slide-out-to-left-1/2 data-[state=closed]:slide-out-to-top-[48%] data-[state=open]:slide-in-from-left-1/2 data-[state=open]:slide-in-from-top-[48%] fixed left-[50%] top-[50%] z-50 grid w-full translate-x-[-50%] translate-y-[-50%] gap-4 border border-grayscale-500 bg-grayscale-800 px-6 text-grayscale-100 shadow-xl duration-200 dark:border-dark-grayscale-500 dark:bg-dark-grayscale-900 dark:text-white sm:max-w-lg sm:rounded-2xl',
            className
          )}
          {...props}
        >
          {children}
          {closeButton && (
            <DialogPrimitive.Close className="absolute right-4 top-6 rounded-sm opacity-70 ring-offset-background transition-opacity hover:opacity-100 focus:outline-none focus:ring-2 focus:ring-ring focus:ring-offset-2 disabled:pointer-events-none data-[state=open]:bg-accent data-[state=open]:text-muted-foreground">
              <X className="h-8 w-8" />
              <span className="sr-only">{t('Close')}</span>
            </DialogPrimitive.Close>
          )}
        </DialogPrimitive.Content>
      </DialogPortal>
    );
  }
);
DialogContent.displayName = DialogPrimitive.Content.displayName;

/**
 * Dialog Header Component
 *
 * Container for dialog title and description with consistent spacing.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Header content (typically DialogTitle and DialogDescription).
 * @param {string} [props.className] - Additional CSS classes.
 *
 * @returns {JSX.Element} Header container element.
 *
 * @example
 * <DialogHeader>
 *   <DialogTitle>Confirm Action</DialogTitle>
 *   <DialogDescription>This action cannot be undone.</DialogDescription>
 * </DialogHeader>
 */
const DialogHeader = ({ className, ...props }) => (
  <div
    className={cn('flex flex-col space-y-1.5 sm:text-left', className)}
    {...props}
  />
);
DialogHeader.displayName = 'DialogHeader';

/**
 * Dialog Footer Component
 *
 * Container for dialog action buttons with consistent spacing and alignment.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Footer content (typically action buttons).
 * @param {string} [props.className] - Additional CSS classes.
 *
 * @returns {JSX.Element} Footer container element.
 *
 * @example
 * <DialogFooter>
 *   <DialogClose asChild>
 *     <Button variant="outline">Cancel</Button>
 *   </DialogClose>
 *   <Button>Confirm</Button>
 * </DialogFooter>
 */
const DialogFooter = ({ className, ...props }) => (
  <div
    className={cn('flex flex-row justify-end space-x-4 pb-8 pt-6', className)}
    {...props}
  />
);
DialogFooter.displayName = 'DialogFooter';

/**
 * Dialog Title Component
 *
 * Accessible title for the dialog. Required for screen readers.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Title text.
 * @param {string} [props.className] - Additional CSS classes.
 * @param {React.Ref} ref - Forwarded ref to the title element.
 *
 * @returns {JSX.Element} Styled heading element.
 *
 * @example
 * <DialogTitle>Delete Account</DialogTitle>
 *
 * @example
 * <DialogTitle className="text-destructive">Warning</DialogTitle>
 */
const DialogTitle = forwardRef(({ className, ...props }, ref) => (
  <DialogPrimitive.Title
    ref={ref}
    className={cn(
      'pb-4 pt-6 text-xl font-extrabold leading-none tracking-tight',
      className
    )}
    {...props}
  />
));
DialogTitle.displayName = DialogPrimitive.Title.displayName;

/**
 * Dialog Description Component
 *
 * Accessible description for the dialog. Provides additional context.
 *
 * @component
 *
 * @param {Object} props - Component properties.
 * @param {React.ReactNode} props.children - Description text.
 * @param {string} [props.className] - Additional CSS classes.
 * @param {React.Ref} ref - Forwarded ref to the description element.
 *
 * @returns {JSX.Element} Styled paragraph element.
 *
 * @example
 * <DialogDescription>
 *   This will permanently delete your account and all associated data.
 * </DialogDescription>
 *
 * @example
 * <DialogDescription className="text-destructive">
 *   Warning: This action cannot be undone.
 * </DialogDescription>
 */
const DialogDescription = forwardRef(({ className, ...props }, ref) => (
  <DialogPrimitive.Description
    ref={ref}
    className={cn('text-sm text-muted-foreground', className)}
    {...props}
  />
));
DialogDescription.displayName = DialogPrimitive.Description.displayName;

export {
  Dialog,
  DialogPortal,
  DialogOverlay,
  DialogClose,
  DialogTrigger,
  DialogContent,
  DialogHeader,
  DialogFooter,
  DialogTitle,
  DialogDescription,
};
