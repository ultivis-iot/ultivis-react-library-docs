import {
  FormControl,
  FormField,
  FormItem,
  FormLabel,
  FormMessage,
  Input,
} from '@components';
import { cn } from '@utils';

/**
 * FormInputField Component
 *
 * A versatile form input field with optional suffix and error handling, designed to integrate with React Hook Form.
 *
 * @component
 *
 * @param {Object} control - The `control` object from React Hook Form for form state management.
 * @param {string} name - The name of the input field in the form schema.
 * @param {string} label - The label displayed above the input field.
 * @param {string} [suffix] - Text or element displayed as a suffix for the input field.
 * @param {string} [suffixPosition='right'] - Position of the suffix (`'left'` or `'right'`).
 * @param {string} [className] - Additional CSS classes for the input field.
 * @param {string} [labelClassName] - Additional CSS classes for the label element.
 * @param {Object} [props] - Extra properties passed to the input element (e.g., `placeholder`, `type`).
 *
 * @example
 * // Example usage
 * <FormInputField
 *   control={control}
 *   name="email"
 *   label="Email Address"
 *   placeholder="Enter your email"
 *   suffix="@example.com"
 *   suffixPosition="right"
 * />
 *
 * @returns {JSX.Element} A styled input field with optional suffix and validation message.
 */
const FormInputField = ({
  control,
  name,
  label,
  suffix,
  suffixPosition = 'right',
  className,
  labelClassName,
  ...props
}) => (
  <FormField
    control={control}
    name={name}
    render={({ field, fieldState: { error } }) => (
      <FormItem>
        {label && <FormLabel className={cn(labelClassName)}>{label}</FormLabel>}
        <div className="relative flex justify-center">
          <FormControl>
            <div className="flex w-full items-center ">
              {suffix && suffixPosition === 'left' && (
                <span className=" text h-14 w-14 rounded-md rounded-e-none border border-grayscale-500 bg-grayscale-500  py-3.5 text-center dark:border-dark-grayscale-500 dark:bg-dark-grayscale-500 dark:text-dark-grayscale-200">
                  {suffix}
                </span>
              )}
              <Input
                {...field}
                {...props}
                value={props?.type === 'file' ? undefined : field.value}
                onChange={(e) => {
                  const files = e.target.files;
                  if (files) {
                    field.onChange(files[0]);
                  } else {
                    field.onChange(e.target.value);
                  }
                }}
                className={cn(
                  `w-full ${
                    error && 'ring-2 ring-destructive focus:outline-destructive'
                  } ${
                    suffix && suffixPosition === 'left' && 'rounded-s-none'
                  } ${
                    suffix && suffixPosition === 'right' && 'rounded-e-none'
                  } px-2 leading-6`,
                  className
                )}
              />
              {suffix && suffixPosition === 'right' && (
                <span className=" text h-14 w-14 rounded-md rounded-s-none border border-grayscale-500 bg-grayscale-500 py-3.5 text-center dark:border-dark-grayscale-500 dark:bg-dark-grayscale-500 dark:text-dark-grayscale-200 ">
                  {suffix}
                </span>
              )}
            </div>
          </FormControl>
        </div>
        {/* TODO : 위젯이 아닌 경우 단독으로 사용할 때 오류 메시지가 나오지 않아 에러 메시지를 추가함 */}
        <FormMessage>{error?.message}</FormMessage>
      </FormItem>
    )}
  />
);

export default FormInputField;
