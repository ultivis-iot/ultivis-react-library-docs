import { useCallback } from 'react';
import { flexRender } from '@tanstack/react-table';
import { useTranslation } from '@hooks';
import { badgeVariantStyles, badgeVariantIcons } from './constants';

/**
 * useCardRenderer Hook
 *
 * Provides card rendering functions for VirtualTable card view.
 *
 * @param {Object} options - Hook options
 * @param {Array} options.columns - TanStack Table column definitions
 * @param {Object} options.table - TanStack Table instance
 * @returns {Object} Card rendering utilities
 */
export const useCardRenderer = ({ columns, table }) => {
  const { t } = useTranslation();

  /**
   * Auto-render card view based on cardConfig in columns
   */
  const renderAutoCard = useCallback(
    (row) => {
      if (!columns) return null;

      // Classify columns with cardConfig by role
      const cardColumns = {
        badge: [],
        title: null,
        description: null,
        meta: [],
        actions: [],
      };

      columns.forEach((col) => {
        if (!col.cardConfig) return;
        const { role } = col.cardConfig;
        const accessorKey = col.accessorKey;
        const value = accessorKey?.includes('.')
          ? accessorKey
              .split('.')
              .reduce((obj, key) => obj?.[key], row.original)
          : row.original[accessorKey];

        const cellContent = col.cell
          ? flexRender(col.cell, {
              getValue: () => value,
              row,
              table,
              column: { id: accessorKey },
              isCardView: true,
            })
          : value;

        if (role === 'badge') {
          const variant =
            col.cardConfig.getVariant?.(value, row) ||
            col.cardConfig.variant ||
            'default';
          cardColumns.badge.push({ value, cellContent, variant, col });
        } else if (role === 'title') {
          cardColumns.title = { value, cellContent, col };
        } else if (role === 'description') {
          cardColumns.description = { value, cellContent, col };
        } else if (role === 'meta') {
          cardColumns.meta.push({ value, cellContent, col });
        } else if (role === 'actions') {
          cardColumns.actions.push({ value, cellContent, col });
        }
      });

      return (
        <div className="flex flex-col gap-2">
          {/* Top: Badge + Actions */}
          {(cardColumns.badge.length > 0 || cardColumns.actions.length > 0) && (
            <div className="flex items-center justify-between">
              <div className="flex items-center gap-2">
                {cardColumns.badge.map((badge, idx) => {
                  const IconComponent =
                    badgeVariantIcons[badge.variant] ||
                    badgeVariantIcons.default;
                  return (
                    <span
                      key={idx}
                      className={`flex items-center gap-2 rounded px-2 ${
                        badgeVariantStyles[badge.variant] ||
                        badgeVariantStyles.default
                      }`}
                    >
                      {IconComponent && (
                        <IconComponent className="my-1 h-6 w-6" />
                      )}
                      {typeof badge.cellContent === 'string'
                        ? t(badge.cellContent)
                        : badge.cellContent}
                    </span>
                  );
                })}
              </div>
              <div className="flex items-center gap-1">
                {cardColumns.actions.map((action, idx) => (
                  <div key={idx}>{action.cellContent}</div>
                ))}
              </div>
            </div>
          )}

          {/* Title */}
          {cardColumns.title && (
            <h3 className="text-base font-semibold text-grayscale-200 dark:text-white">
              {cardColumns.title.cellContent}
            </h3>
          )}

          {/* Description */}
          {cardColumns.description && (
            <p className="line-clamp-2 text-sm text-grayscale-400">
              {cardColumns.description.cellContent}
            </p>
          )}

          {/* Meta info */}
          {cardColumns.meta.length > 0 && (
            <div className="mt-1 flex items-center gap-2 text-xs text-grayscale-500">
              {cardColumns.meta.map((meta, idx) => (
                <span key={idx}>
                  {idx > 0 && <span className="mr-2">|</span>}
                  {meta.cellContent}
                </span>
              ))}
            </div>
          )}
        </div>
      );
    },
    [columns, table, t]
  );

  /**
   * Default list view rendering when no cardConfig (for mobile)
   */
  const renderDefaultListView = useCallback(
    (row) => {
      if (!columns) return null;

      // Collect cell data to render
      const cellsData = columns
        .map((col) => {
          const accessorKey = col.accessorKey;
          const columnId = col.id || accessorKey;

          const value = accessorKey
            ? accessorKey.includes('.')
              ? accessorKey
                  .split('.')
                  .reduce((obj, key) => obj?.[key], row.original)
              : row.original[accessorKey]
            : null;

          const cellContent = col.cell
            ? flexRender(col.cell, {
                getValue: () => value,
                row,
                table,
                column: { id: columnId },
                isCardView: true,
              })
            : value;

          const headerText =
            typeof col.header === 'string'
              ? col.header
              : typeof col.header === 'function'
                ? accessorKey || columnId
                : accessorKey || columnId;

          return {
            key: columnId,
            header: headerText,
            value,
            cellContent,
            hideLabel: headerText === '',
            hidden: headerText === null || headerText === undefined,
          };
        })
        .filter((cell) => !cell.hidden);

      const visibleCells = cellsData.filter(
        (cell) =>
          cell.hideLabel ||
          (cell.value !== null && cell.value !== undefined && cell.value !== '')
      );

      return (
        <div className="flex flex-col gap-3 pr-6">
          {visibleCells.map((cell, idx) => (
            <div key={cell.key || idx} className="flex items-start">
              {!cell.hideLabel && (
                <span className="w-[72px] flex-shrink-0 pt-0.5 text-xs leading-5 text-grayscale-400 dark:text-white/[38%]">
                  {t(cell.header)}
                </span>
              )}
              <span className="min-w-0 flex-1 text-sm leading-5 text-grayscale-200 dark:text-white">
                {cell.cellContent}
              </span>
            </div>
          ))}
        </div>
      );
    },
    [columns, table, t]
  );

  /**
   * Check if any column has cardConfig
   */
  const hasCardConfig = columns?.some((col) => col.cardConfig);

  return {
    renderAutoCard,
    renderDefaultListView,
    hasCardConfig,
  };
};

export default useCardRenderer;
