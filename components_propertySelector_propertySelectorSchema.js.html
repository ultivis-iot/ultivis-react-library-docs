import { z } from 'zod';
import { dataSchema } from '@components';

const computedInventoryList = [
  'lastDeviceMessage',
  'childDevicesCount',
  'childAssetsCount',
  'configurationSnapshot',
];

const computedDetail = z.object({
  type: z.enum(['alarm', 'event', 'measurement']),
  detail: dataSchema,
});

/**
 * `propertySchema`
 *
 * A Zod schema to validate property configurations, including computed and complex properties.
 * The schema ensures that computed properties have appropriate details defined unless they belong to a predefined inventory list.
 *
 * @constant
 * @type {z.ZodObject}
 *
 * @property {string} path - The path of the property.
 * @property {string} label - The label of the property.
 * @property {boolean} computed - Indicates if the property is computed.
 * @property {boolean} complex - Indicates if the property is complex.
 * @property {boolean} show - Determines if the property should be shown.
 * @property {string} targetPath - The target path of the property.
 * @property {object} [computedDetail] - Optional details for computed properties, validated by `computedDetail` schema.
 */
export const propertySchema = z
  .object({
    path: z.string(),
    label: z.string(),
    computed: z.boolean(),
    complex: z.boolean(),
    show: z.boolean(),
    targetPath: z.string(),
    computedDetail: computedDetail.optional(),
  })
  .refine((data) => {
    if (data.computed && !computedInventoryList.includes(data.path)) {
      return data.computedDetail !== undefined;
    }
    return true;
  });
