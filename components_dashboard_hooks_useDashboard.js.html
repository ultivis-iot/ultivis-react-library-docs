/**
 * ContextDashboard - c8y-specific dashboard management hook using u5s_Dashboard structure.
 * @see useDynamicDashboard for platform-independent version
 */
import { cloneDeep } from 'lodash-es';
import { useRouting } from '@hooks';
import { useDashboardApi } from '../context';
import { useDashboardStore } from '../store/useDashboardStore';
import {
  getBreakpoints,
  getCols,
  getWindowSize,
  calculateBreakpoint as calcBreakpoint,
  getWidgetWidth as getWidgetW,
} from '../utils/gridUtils';

/**
 * Custom hook for managing dashboards and widgets.
 * c8y-specific version that handles u5s_Dashboard data structure.
 *
 * @returns {Object} Dashboard management object containing:
 * @returns {Object} breakpoints - Responsive breakpoint configuration
 * @returns {Object} cols - Column configuration for different breakpoints
 * @returns {Function} postDashboard - Creates a new dashboard
 * @returns {Function} putDashboard - Updates an existing dashboard
 * @returns {Function} postWidget - Adds a new widget to the dashboard
 * @returns {Function} putWidget - Edits or removes a widget from the dashboard
 * @returns {Function} putLayout - Updates the layout of widgets
 * @returns {Function} calculateBreakpoint - Calculates breakpoint and columns for given width
 * @returns {Function} getWidgetWidth - Gets widget width based on max columns
 */

export const useDashboard = () => {
  const { appName } = useRouting();
  const { createDashboard, createChildDashboard } = useDashboardApi();
  const { containerWidth, draftDashboard, setDraftDashboard, isFrozen } =
    useDashboardStore((state) => ({
      containerWidth: state.containerWidth,
      draftDashboard: state.draftDashboard,
      setDraftDashboard: state.setDraftDashboard,
      isFrozen: state.isFrozen,
    }));

  const breakpoints = getBreakpoints(isFrozen);
  const cols = getCols(isFrozen);

  const methods = {
    /**
     * Creates a new dashboard.
     * @param {Object} params - Dashboard parameters
     * @param {number} params.widgetMargin - Margin between widgets
     * @param {string} params.headerStyle - Header styling class
     * @param {number} params.priority - Dashboard priority
     * @param {string} params.name - Dashboard name
     * @param {string} params.icon - Dashboard icon
     * @param {string} params.type - Dashboard type
     * @param {boolean} params.global - Whether dashboard is global
     * @param {string} params.category - Dashboard category
     * @param {string} params.sourceId - Source identifier
     * @param {Object} params.period - Dashboard period configuration
     * @returns {Promise<Object>} Created dashboard response
     */
    postDashboard: async ({
      widgetMargin,
      headerStyle,
      priority,
      name,
      icon,
      type,
      global,
      category,
      sourceId,
      period,
    }) => {
      const dashboardObject = {
        u5s_Dashboard: {
          name,
          priority,
          icon,
          period,
          widgetClasses: {
            [headerStyle]: true,
          },
          widgetMargin,
        },
      };
      dashboardObject.name = name;

      if (category === 'device') {
        dashboardObject.u5s_Dashboard.deviceTypeValue = type;
      }

      // 모든 디바이스에 대시보드 적용
      if (global) {
        dashboardObject.c8y_Global = {};
        dashboardObject.u5s_Dashboard.deviceType = true;
        dashboardObject.u5s_Dashboard.children = {};
        dashboardObject[`u5s_Dashboard!type!${type}`] = {};

        return await createDashboard(dashboardObject);
      }

      if (!global && sourceId) {
        dashboardObject[`u5s_Dashboard!${category}!${sourceId}`] = {};
        return await createChildDashboard(dashboardObject, sourceId);
      }

      dashboardObject.c8y_Global = {};
      dashboardObject[`u5s_Dashboard!name!home-${appName}`] = {};

      return await createDashboard(dashboardObject);
    },

    /**
     * Updates an existing dashboard with new details.
     * @param {Object} params - Update parameters
     * @param {number} params.widgetMargin - Margin between widgets
     * @param {string} params.headerStyle - Header styling class
     * @param {number} params.priority - Dashboard priority
     * @param {string} params.name - Dashboard name
     * @param {string} params.icon - Dashboard icon
     * @param {Object} params.period - Dashboard period configuration
     * @returns {Object} Updated dashboard in store
     */
    putDashboard: ({
      widgetMargin,
      headerStyle,
      priority,
      name,
      icon,
      period,
    }) => {
      const cloneDashboard = cloneDeep(draftDashboard);
      cloneDashboard.name = name;
      cloneDashboard.u5s_Dashboard.priority = priority;
      cloneDashboard.u5s_Dashboard.icon = icon;
      cloneDashboard.u5s_Dashboard.widgetMargin = widgetMargin;
      cloneDashboard.u5s_Dashboard.name = name;
      cloneDashboard.u5s_Dashboard.period = period;

      delete cloneDashboard.u5s_Dashboard.widgetClasses;
      cloneDashboard.u5s_Dashboard.widgetClasses = { [headerStyle]: true };

      return setDraftDashboard(cloneDashboard);
    },

    /**
     * Adds a new widget to the dashboard.
     * @param {Object} params - Widget parameters
     * @param {Object} params.widget - Widget object to add
     * @returns {Object} Updated dashboard in store
     */
    postWidget: ({ widget }) => {
      const cloneDashboard = cloneDeep(draftDashboard);
      // id 생성
      const id = String(Math.random()).substring(2);
      const widgets = Object.values(
        cloneDashboard?.u5s_Dashboard?.children ?? []
      );

      const { maxYPosition, maxRowIndex } = widgets.reduce(
        (acc, widget) => {
          return {
            maxYPosition: Math.max(acc.maxYPosition, widget.y),
            maxRowIndex: Math.max(acc.maxRowIndex, widget.y + widget.h),
          };
        },
        { maxYPosition: 0, maxRowIndex: 0 }
      );

      // 현재 행에 배치된 위젯의 넓이 체크크
      const maxXPositionInRow = widgets.reduce((maxX, widget) => {
        return widget.y === maxYPosition
          ? Math.max(maxX, widget.x + widget.w)
          : maxX;
      }, 0);

      const isOver = maxXPositionInRow + 4 <= getWindowSize(containerWidth);

      const widgetObject = {
        ...widget,
        i: id,
        x: isOver ? maxXPositionInRow : 0,
        y: isOver ? maxYPosition : maxRowIndex,
        w: 4,
        h: 4,
      };

      // 복사한 대시보드에 위젯 추가
      cloneDashboard.u5s_Dashboard.children = {
        ...cloneDashboard.u5s_Dashboard.children,
        [id]: widgetObject,
      };

      return setDraftDashboard(cloneDashboard);
    },

    /**
     * Edits or removes a widget from the dashboard.
     * @param {Object} params - Widget management parameters
     * @param {Object} params.widget - Updated widget object (for editing)
     * @param {string} params.widgetId - Widget ID to edit or remove
     * @param {string} params.action - Action type: 'edit' | 'remove'
     * @returns {Object} Updated dashboard in store
     */
    putWidget: ({ widget, widgetId, action }) => {
      const cloneDashboard = cloneDeep(draftDashboard);

      if (action === 'edit') {
        cloneDashboard.u5s_Dashboard.children = {
          ...cloneDashboard.u5s_Dashboard.children,
          [widgetId]: widget,
        };
      }

      if (action === 'remove') {
        delete cloneDashboard.u5s_Dashboard.children[widgetId];
      }

      return setDraftDashboard(cloneDashboard);
    },

    /**
     * Updates the layout of widgets in the dashboard.
     * @param {Array<Object>} newLayout - Layout objects with widget position and size
     * @param {string} newLayout[].i - Widget ID
     * @param {number} newLayout[].x - X position in grid
     * @param {number} newLayout[].y - Y position in grid
     * @param {number} newLayout[].w - Width in grid units
     * @param {number} newLayout[].h - Height in grid units
     * @returns {Object} Updated dashboard in store
     */
    putLayout: (newLayout) => {
      const updatedDashboard = cloneDeep(draftDashboard);

      newLayout.forEach(({ i, x, y, w, h }) => {
        const widget = updatedDashboard.u5s_Dashboard.children[i];
        if (widget) {
          updatedDashboard.u5s_Dashboard.children[i] = {
            ...widget,
            x,
            y,
            w,
            h,
          };
        }
      });

      return setDraftDashboard(updatedDashboard);
    },

    /**
     * Calculates breakpoint and columns based on container width.
     * NOTE: 편집 모드시에 무조건 cols 12
     * @param {number} width - Container width in pixels
     * @returns {Object} Object containing breakpoint name and column count
     * @returns {string} breakpoint - Breakpoint identifier (xl, lg, md, sm, xs, xss)
     * @returns {number} cols - Number of columns for the breakpoint
     */
    calculateBreakpoint: (width) => calcBreakpoint(width, isFrozen),

    /**
     * Gets the appropriate widget width based on maximum columns.
     * @param {number} maxCols - Maximum number of columns available
     * @param {Object} widget - Widget object
     * @param {number} widget.w - Widget's desired width
     * @returns {number} Actual widget width (limited by maxCols)
     */
    getWidgetWidth: getWidgetW,
  };

  return { breakpoints, cols, ...methods };
};
