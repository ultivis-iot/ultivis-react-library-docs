import {
  useEffect,
  useReducer,
  useMemo,
  useContext,
  createContext,
} from 'react';
import { startOfDay, sub } from 'date-fns';
import { WidgetDatePicker } from '@components';
import { useRouting } from '@hooks';

// Context for sharing dashboard date state globally
const DashboardDateContext = createContext(null);

// Action types used in the reducer
const ActionTypes = {
  SET_DATE_RANGE: 'SET_DATE_RANGE', // Set date range
  RESET_DATE_RANGE: 'RESET_DATE_RANGE', // Reset date range
  EDIT_CONFIG: 'EDIT_CONFIG', // Edit configuration
  APPLY_FILTER: 'APPLY_FILTER', // Apply filter
  REFETCHED: 'REFETCHED', // Refetch data
  TOGGLE_REALTIME: 'TOGGLE_REALTIME', // Toggle realtime mode
  SET_AGGREGATION_TYPE: 'SET_AGGREGATION_TYPE', // Set aggregation type
  SET_DATAPOINT: 'SET_DATAPOINT', // Set data point
  SET_PERIOD: 'SET_PERIOD', // Set period
  RESET: 'RESET', // Reset entire state
};

/**
 * Calculates start and end dates based on the given period
 * @param {Object} params - Parameter object
 * @param {string} params.period - Period type ('minutes', 'hours', '24hours', 'days', 'weeks', 'months', 'years')
 * @returns {Object} { startDate, endDate } - Calculated date range
 */
const calculateDateRange = ({ period }) => {
  const now = new Date();
  let startDate, endDate;
  switch (period) {
    case 'minutes':
    case 'hours':
    case '24hours': {
      // Minutes/Hours: From n minutes/hours ago to now
      const unit = period === '24hours' ? 'hours' : period;
      const amount = period === '24hours' ? 24 : 1;
      startDate = sub(now, { [unit]: amount });
      endDate = now;
      break;
    }
    case 'days':
      // Days: From start of today (00:00) to now
      startDate = startOfDay(now);
      endDate = now;
      break;
    case 'weeks':
    case 'months':
    case 'years':
      // Weeks/Months/Years: From start of 1 week/month/year ago to now
      startDate = startOfDay(sub(now, { [period]: 1 }));
      endDate = now;
      break;
    default:
      // Default: From 1 hour ago to now
      startDate = sub(now, { hours: 1 });
      endDate = now;
      break;
  }

  return { startDate, endDate };
};

// Reducer action handler map
const reducerHandlers = {
  // Set date range directly and increment refetchCount
  [ActionTypes.SET_DATE_RANGE]: (state, action) => ({
    ...state,
    dateRange: action.payload,
    refetchCount: state.refetchCount + 1,
  }),
  // Reset date range by recalculating based on current period
  [ActionTypes.RESET_DATE_RANGE]: (state, action, calculateDateRange) => ({
    ...state,
    dateRange: calculateDateRange({
      period: state.period,
    }),
    refetchCount: state.refetchCount + 1,
    ...action.payload,
  }),
  // Edit configuration and disable realtime mode
  [ActionTypes.EDIT_CONFIG]: (state, action, calculateDateRange) => ({
    ...state,
    dateRange: calculateDateRange({ ...action.payload }),
    refetchCount: state.refetchCount + 1,
    isRealtime: false,
    ...action.payload,
  }),
  // Increment refetchCount and recalculate date range if in realtime mode
  [ActionTypes.REFETCHED]: (state, action, calculateDateRange) => ({
    ...state,
    dateRange: state.isRealtime
      ? calculateDateRange({ period: state.period })
      : state.dateRange,
    refetchCount: state.refetchCount + 1,
  }),
  // Toggle realtime mode and increment refetchCount
  [ActionTypes.TOGGLE_REALTIME]: (state, action) => ({
    ...state,
    refetchCount: state.refetchCount + 1,
    isRealtime: !state.isRealtime,
    ...action.payload,
  }),
  // Change aggregation type and increment refetchCount
  [ActionTypes.SET_AGGREGATION_TYPE]: (state, action) => ({
    ...state,
    aggregationType: action.payload,
    refetchCount: state.refetchCount + 1,
  }),
  // Change period and recalculate date range accordingly
  [ActionTypes.SET_PERIOD]: (state, action, calculateDateRange) => ({
    ...state,
    period: action.payload,
    dateRange: calculateDateRange({
      period: action.payload,
    }),
    refetchCount: state.refetchCount + 1,
  }),
  // Update data point related settings and increment refetchCount
  [ActionTypes.SET_DATAPOINT]: (state, action) => ({
    ...state,
    refetchCount: state.refetchCount + 1,
    ...action.payload,
  }),
  // Completely reset state (used when switching dashboards)
  [ActionTypes.RESET]: (state, action) => ({
    isRealtime: action.payload.isRealtime ?? true,
    period: action.payload.period ?? 'hours',
    dateRange: calculateDateRange({
      period: action.payload.period ?? 'hours',
    }),
    aggregationType: action.payload.aggregationType,
    refetchCount: 0,
  }),
};

/**
 * Main reducer function that updates state
 * @param {Object} state - Current state
 * @param {Object} action - Dispatched action { type, payload }
 * @param {Function} calculateDateRange - Date range calculation function
 * @returns {Object} New state
 */
const reducer = (state, action, calculateDateRange) => {
  const handler = reducerHandlers[action.type];
  if (!handler)
    throw new Error('Unhandled action type in useWidgetState reducer');
  return handler(state, action, calculateDateRange);
};

/**
 * Provider component that provides dashboard date state
 * @param {Object} props
 * @param {ReactNode} props.children - Child components
 * @param {string} props.initialPeriod - Initial period (default: 'days')
 * @param {boolean} props.initialRealtime - Initial realtime mode (default: true)
 * @param {string} props.initialAggregationType - Initial aggregation type (default: 'none')
 */
export const DashboardDateProvider = ({
  children,
  initialPeriod = 'days',
  initialRealtime = true,
  initialAggregationType = 'none',
}) => {
  const { dashboardId } = useRouting();

  // State management using useReducer
  const [state, dispatch] = useReducer(
    (s, a) => reducer(s, a, calculateDateRange),
    null,
    () => ({
      dateRange: calculateDateRange({
        period: initialPeriod,
      }),
      isRealtime: initialRealtime,
      period: initialPeriod,
      aggregationType: initialAggregationType,
      refetchCount: 0,
    })
  );

  const { dateRange, isRealtime, refetchCount, aggregationType, period } =
    state;

  // Handler to refetch data
  const handleRefetch = () => {
    dispatch({ type: ActionTypes.REFETCHED });
  };

  // Reset state when switching dashboards
  useEffect(() => {
    dispatch({
      type: ActionTypes.RESET,
      payload: {
        period: initialPeriod,
        isRealtime: initialRealtime,
        aggregationType: initialAggregationType,
      },
    });
  }, [dashboardId, initialPeriod, initialRealtime, initialAggregationType]);

  // Memoize context value to prevent unnecessary re-renders
  const value = useMemo(
    () => ({
      state,
      dispatch,
      refetchCount,
      dateRange,
      isRealtime,
      aggregationType,
      period,
      calculateDateRange,
    }),
    [
      state,
      dispatch,
      refetchCount,
      dateRange,
      isRealtime,
      aggregationType,
      period,
    ]
  );

  return (
    <DashboardDateContext.Provider value={value}>
      <div className="mx-2 my-2 lg:mx-5">
        <WidgetDatePicker
          className="overflow-hidden"
          period={period}
          dispatch={dispatch}
          dateRange={dateRange}
          aggregationType={aggregationType}
          onRefetch={handleRefetch}
          useRealtime={true}
          isRealtime={isRealtime}
          showDatePicker={!isRealtime}
          storeCache={true}
          storeKey={`dashboard_${dashboardId}`}
        />
      </div>
      {children}
    </DashboardDateContext.Provider>
  );
};

/**
 * Custom hook to access dashboard date state
 * @returns {Object} { state, dispatch, refetchCount, dateRange, isRealtime, aggregationType, period, calculateDateRange }
 * @throws {Error} Throws error if used outside DashboardDateProvider
 */
export const useDashboardDate = () => {
  const context = useContext(DashboardDateContext);
  if (!context)
    throw new Error(
      'useDashboardDate must be used within a DashboardDateProvider'
    );
  return context;
};
