import { useCallback, useState } from 'react';
import { flexRender } from '@tanstack/react-table';
import { Filter } from 'lucide-react';
import { Button, EmptyState, LoadingSpinner } from '@components';
import { useTranslation } from '@hooks';
import { useCardRenderer } from './CardRenderer';
import FilterModal from './FilterModal';
import RowActions from './RowActions';

/**
 * VirtualTableContent Component
 *
 * Core rendering logic for VirtualTable. Handles both table and card view rendering.
 *
 * @component
 * @internal
 */
const VirtualTableContent = ({
  data,
  pageData,
  columns,
  hasNextPage,
  isFetching,
  onRowClick,
  rowActions = [],
  emptyMessage,
  classNames = {},
  renderActionIcon,
  sortIcons = { asc: ' ðŸ”¼', desc: ' ðŸ”½' },
  renderCard,
  tableType = 'standard',
  tableHookResult,
  enableFiltering = false,
  columnFilters = [],
  onColumnFiltersChange,
}) => {
  const { t } = useTranslation();
  const [isFilterModalOpen, setIsFilterModalOpen] = useState(false);
  const { resolvedViewMode } = tableHookResult;

  const setColumnFilters = onColumnFiltersChange ?? (() => {});

  const tableData = tableType === 'infinite' ? pageData : data;

  const {
    table,
    rowVirtualizer,
    tableContainerRef,
    virtualRows,
    rows,
    observerTarget,
  } = tableHookResult;

  const { renderAutoCard, renderDefaultListView, hasCardConfig } =
    useCardRenderer({ columns, table });

  // Row click handler
  const handleRowClick = useCallback(
    (row, event) => {
      if (!onRowClick) return;
      if (event.target.closest('[data-row-action]')) return;
      onRowClick(row.original, row);
    },
    [onRowClick]
  );

  // measureElement callback
  const measureElementCallback = useCallback(
    (node) => {
      if (node && rowVirtualizer?.measureElement) {
        rowVirtualizer.measureElement(node);
      }
    },
    [rowVirtualizer]
  );

  const hasActiveFilters = columnFilters.length > 0;

  // Effective card renderer (custom > auto card > default list)
  const effectiveRenderCard = renderCard
    ? renderCard
    : resolvedViewMode === 'card'
      ? hasCardConfig
        ? renderAutoCard
        : renderDefaultListView
      : null;

  // Filter button UI rendering
  const renderFilterButton = () => {
    if (!enableFiltering) return null;

    return (
      <div className="flex items-center justify-end">
        <Button
          variant="ghost"
          size="sm"
          title={t('Filter')}
          className={
            hasActiveFilters
              ? 'text-primary dark:text-dark-primary'
              : 'text-grayscale-300 dark:text-white'
          }
          onClick={() => setIsFilterModalOpen(true)}
        >
          <Filter className="h-4 w-4" />
          {hasActiveFilters && (
            <span className="ml-1 text-xs">({columnFilters.length})</span>
          )}
        </Button>
      </div>
    );
  };

  if (!tableData?.length && tableType === 'standard') {
    return (
      <div className={`h-full min-h-0 w-full ${classNames.container ?? ''}`}>
        <div
          ref={tableContainerRef}
          className={`custom-scroll h-full w-full overflow-auto bg-grayscale-1000 dark:bg-dark-bg-333 ${classNames.scrollContainer ?? ''}`}
        >
          {enableFiltering && (
            <div className="sticky top-0 z-20 bg-grayscale-1000 dark:bg-dark-bg-333">
              {renderFilterButton()}
            </div>
          )}
          <div className="flex h-full items-center justify-center">
            <EmptyState message={emptyMessage || t('No items to display')} />
          </div>
        </div>
      </div>
    );
  }

  return (
    <div className={`h-full min-h-0 w-full ${classNames.container ?? ''}`}>
      {/* Filter modal */}
      {enableFiltering && isFilterModalOpen && (
        <FilterModal
          table={table}
          open={isFilterModalOpen}
          onClose={() => setIsFilterModalOpen(false)}
          onSave={(filters) => setColumnFilters(filters)}
        />
      )}

      <div
        ref={tableContainerRef}
        className={`custom-scroll h-full w-full overflow-auto bg-grayscale-1000 dark:bg-dark-bg-333 ${classNames.scrollContainer ?? ''}`}
      >
        {tableData?.length > 0 ? (
          <div className="grid w-full">
            {/* Sticky header area */}
            <div className="sticky top-0 z-20 bg-grayscale-1000 dark:bg-dark-bg-333">
              {/* Card view: filter button on separate row */}
              {effectiveRenderCard && renderFilterButton()}
              {/* Table view: header with filter button */}
              {!effectiveRenderCard && (
                <div
                  className={`relative flex items-center px-4 py-2 ${classNames.header ?? ''}`}
                >
                  <div className="flex w-full">
                    {table.getHeaderGroups().map((headerGroup) =>
                      headerGroup.headers.map((header) => {
                        const hasSize = header.column.columnDef.size !== 0;
                        const align =
                          header.column.columnDef.meta?.headerAlign ?? 'left';
                        const alignClass = {
                          left: 'justify-start',
                          center: 'justify-center',
                          right: 'justify-end',
                        }[align];
                        return (
                          <div
                            key={header.id}
                            style={{
                              width:
                                header.getSize() !== 0
                                  ? header.getSize()
                                  : '100%',
                              minWidth: '32px',
                              flex: hasSize
                                ? 'unset'
                                : header.column.columnDef.flex || 1,
                            }}
                            className={`flex text-xs text-grayscale-400 dark:text-white/[38%] ${alignClass}`}
                          >
                            <div
                              className={
                                header.column.getCanSort()
                                  ? 'cursor-pointer select-none'
                                  : ''
                              }
                              onClick={header.column.getToggleSortingHandler()}
                            >
                              {typeof header.column.columnDef.header ===
                              'string'
                                ? t(header.column.columnDef.header)
                                : flexRender(
                                    header.column.columnDef.header,
                                    header.getContext()
                                  )}
                              {sortIcons[header.column.getIsSorted()] ?? null}
                            </div>
                          </div>
                        );
                      })
                    )}
                  </div>
                  {/* Filter button - absolute positioned */}
                  {enableFiltering && (
                    <div className="absolute right-2 top-1/2 -translate-y-1/2">
                      <Button
                        variant="ghost"
                        size="sm"
                        title={t('Filter')}
                        className={
                          hasActiveFilters
                            ? 'text-primary dark:text-dark-primary'
                            : 'text-grayscale-300 dark:text-white'
                        }
                        onClick={() => setIsFilterModalOpen(true)}
                      >
                        <Filter className="h-4 w-4" />
                        {hasActiveFilters && (
                          <span className="ml-1 text-xs">
                            ({columnFilters.length})
                          </span>
                        )}
                      </Button>
                    </div>
                  )}
                </div>
              )}
            </div>
            <div
              className="relative grid w-full px-2"
              style={{
                height: `${rowVirtualizer?.getTotalSize()}px`,
              }}
            >
              {virtualRows?.map((virtualRow) => {
                const row = rows?.[virtualRow.index];
                if (!row) return null;

                // Card view rendering
                if (effectiveRenderCard) {
                  return (
                    <div
                      data-index={virtualRow.index}
                      ref={measureElementCallback}
                      key={row.id}
                      className={`group absolute w-full pb-2 ${classNames.row ?? ''}`}
                      style={{
                        transform: `translateY(${virtualRow.start}px)`,
                      }}
                    >
                      <div
                        className={`relative h-full rounded-lg border border-grayscale-700 p-4 text-grayscale-200 dark:border-white/[8%] dark:text-white ${
                          onRowClick
                            ? 'cursor-pointer hover:border-primary hover:dark:border-dark-primary'
                            : ''
                        }`}
                        onClick={
                          onRowClick ? (e) => handleRowClick(row, e) : undefined
                        }
                      >
                        {/* RowActions button */}
                        {rowActions.length > 0 && (
                          <div className="absolute right-2 top-2 z-10">
                            <div data-row-action>
                              <RowActions
                                row={row}
                                rowActions={rowActions}
                                renderActionIcon={renderActionIcon}
                              />
                            </div>
                          </div>
                        )}
                        {renderCard
                          ? renderCard(row, table)
                          : effectiveRenderCard(row)}
                      </div>
                    </div>
                  );
                }

                // Default table view rendering
                const rowDepth = row.depth ?? 0;
                const depthIndent = rowDepth * 24;

                return (
                  <div
                    data-index={virtualRow.index}
                    ref={measureElementCallback}
                    key={row.id}
                    className={`group absolute w-full pb-2 ${classNames.row ?? ''}`}
                    style={{
                      transform: `translateY(${virtualRow.start}px)`,
                    }}
                  >
                    <div
                      className={`flex h-full items-center gap-y-2 rounded-lg border border-grayscale-700 px-4 py-3 text-grayscale-200 dark:border-white/[8%] dark:text-white ${
                        onRowClick
                          ? 'cursor-pointer hover:text-primary hover:dark:text-dark-primary'
                          : ''
                      }`}
                      onClick={
                        onRowClick ? (e) => handleRowClick(row, e) : undefined
                      }
                    >
                      {row.getVisibleCells().map((cell) => {
                        const cellSize = cell.column.getSize();
                        const shouldIndent =
                          cell.column.columnDef.meta?.indentCell === true;
                        const align =
                          cell.column.columnDef.meta?.cellAlign ?? 'left';
                        const alignClass = {
                          left: 'justify-start',
                          center: 'justify-center',
                          right: 'justify-end',
                        }[align];
                        return (
                          <div
                            key={cell.id}
                            style={{
                              maxWidth: cellSize !== 0 ? cellSize : '100%',
                              minWidth: '32px',
                              flex: cell.column.columnDef.flex || 1,
                              overflowWrap: 'anywhere',
                              paddingLeft: shouldIndent
                                ? `${depthIndent}px`
                                : undefined,
                            }}
                            className={`flex w-full text-sm font-medium ${alignClass} ${classNames.cell ?? ''}`}
                          >
                            {flexRender(cell.column.columnDef.cell, {
                              ...cell.getContext(),
                              isCardView: false,
                            })}
                          </div>
                        );
                      })}

                      {/* RowActions separated */}
                      {rowActions.length > 0 && (
                        <div className="absolute right-2 z-10">
                          <div data-row-action>
                            <RowActions
                              row={row}
                              rowActions={rowActions}
                              renderActionIcon={renderActionIcon}
                            />
                          </div>
                        </div>
                      )}
                    </div>
                  </div>
                );
              })}
            </div>
          </div>
        ) : (
          <div className="flex h-full items-center justify-center">
            <EmptyState message={emptyMessage || t('No items to display')} />
          </div>
        )}

        {/* Infinite scroll loading state */}
        {tableType === 'infinite' &&
          observerTarget &&
          (hasNextPage || isFetching) && (
            <div
              ref={observerTarget}
              className="relative flex h-12 items-center justify-center"
            >
              {isFetching && <LoadingSpinner size="sm" withText={false} />}
            </div>
          )}
      </div>
    </div>
  );
};

export default VirtualTableContent;
